/** Objeto Matriz */
class Matriz {
	/**
	 * Constructor de Matriz
	 * @param {number} d - Valor de la dimension
	 * @param {Nodo} c - Nodo inicial (Posicion superior izquierda)
	 * @param {Array} s - Array que contiene la solucion del laberinto
	 */
    constructor(d = 0, c = null, s = null) {
        this.control = c;
        this.solucion = s;
        this.dimension = d;
    }
}
